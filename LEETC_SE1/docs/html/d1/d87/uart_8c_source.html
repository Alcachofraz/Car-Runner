101 static uint8_t UART_SetDivisors(uint32_t baud) {
102  if (baud > 115200) return -1;
103  if (baud == 0) baud = 9600;
104 
105  int pclk = UART_GetPCLK();
106 
107  double d = (double) pclk / (16 * baud); // 27,1267 with PCLK = 50 MHz and baud = 115200
108  int i = (int) d;
109 
110 
111  // Best results:
112  int best[2] = {0, 1}; // Pre-scaler values, divisor and multiplier (default is 0, 1)
113  double bestMargin = 0xFFFFFFFF; // Best margin between real fraction and pre-scaler generated margin
114  int bestIntVal = 0; // Best integer value
115 
116  double currentFraction;
117  double currentMargin;
118 
119  // i * (1 + fraction) = d <=> fraction = (d - i) / i
120  for (int intVal = i/2; intVal <= i; intVal++) {
121  currentFraction = (double) (d - intVal) / intVal;
122  for (int divAddVal = 0; divAddVal <= 14; divAddVal++) {
123  for (int mulVal = divAddVal + 1; mulVal <= 15; mulVal++) {
124  if ((currentMargin = (double) MARGIN(currentFraction, (double) divAddVal/mulVal)) < bestMargin) {
125  best[0] = divAddVal;
126  best[1] = mulVal;
127  bestMargin = currentMargin;
128  bestIntVal = intVal;
129  }
130  }
131  }
132  }
133 
134  //printf("Asked Baud Rate: %f\nReal Baud Rate: %f\n", (double) baud, (double) pclk / (16 * bestIntVal * (1 + ((double) best[0] / best[1]))));
135 
136  UARTx->LCR |= UART_LCR_DLAB_EN; // Set DLAB
137 
138  // Clean registers:
139  UARTx->DLL &= ~(0xFF);
140  UARTx->DLM &= ~(0xFF);
141  UARTx->FDR &= ~(0xFF);
142 
143  // Write values:
144  UARTx->DLL = UART_LOAD_DLL(bestIntVal);
145  UARTx->DLM = UART_LOAD_DLM(bestIntVal);
146  UARTx->FDR = UART_LOAD_FDR(best[0], best[1]);
147 
148  UARTx->LCR &= ~UART_LCR_DLAB_EN; // Clear DLAB
149  return 0;
150 }

219  UARTx->IER &= (~UART_IER_RBRINT_EN) & UART_IER_BITMASK; // Disable RBRINT Interrupts

222  rbuffer.rx[rbuffer.rxWrite] = UART_RBR_ReadChar();

228 static void UART_IntTransmit() {
229  while (UART_CheckBusy()); // Wait until UART isn't busy
230 
231  UARTx->IER &= (~UART_IER_THREINT_EN) & UART_IER_BITMASK;
232 
233  while (!RBUF_IS_EMPTY(rbuffer.txWrite, rbuffer.txRead)
234  && ((UARTx->LSR & UART_LSR_THRE) != 0)) {
235  // Write next Ring Buffer char to THR:
236  UART_THR_WriteChar((unsigned char) rbuffer.tx[rbuffer.txRead]);
237  RBUF_INCR(rbuffer.txRead);
238  }
239 
240  if (RBUF_IS_EMPTY(rbuffer.txWrite, rbuffer.txRead)) { // Ring Buffer is empty:
241  intrTxStatus = false;
242  UARTx->IER &= (~UART_IER_THREINT_EN) & UART_IER_BITMASK; // Disable THRE Interrupts
243  }
244  else { // Ring Buffer still has data:
245  intrTxStatus = true;
246  UARTx->IER |= UART_IER_THREINT_EN & UART_IER_BITMASK; // Re-enable THRE Interrupts
247  }
248 }

257 void UART2_IRQHandler(void) {
258  uint32_t iir, lsr;
259  while (!((iir = UARTx->IIR) & UART_IIR_INTSTAT_PEND)) {
260  switch (iir & UART_IIR_INTID_MASK) {
261  case UART_IIR_INTID_RLS: // Error of some kind:
262  // Check line status
263  lsr = UARTx->LSR;
264  // Mask out the Receive Ready and Transmit Holding empty status
265  lsr &= (UART_LSR_OE | UART_LSR_PE | UART_LSR_FE | UART_LSR_BI | UART_LSR_RXFE);
266  // If any error exist
267  if (lsr) UART_ErrorHandler(lsr);
268  break;
269  case UART_IIR_INTID_RDA:
270  case UART_IIR_INTID_CTI: // Can Read:
271  UART_IntReceive();
272  break;
273  case UART_IIR_INTID_THRE: // Can Write:
274  UART_IntTransmit();
275  break;
276  }
277  }

280 bool UART_Init(uint32_t baud) {
281  WAIT_Init(SYS);
282 
283  uint8_t tmp;
284 
285  UARTx = LPC_UART2; // We will use UART2
286 
287  LPC_PINCON->PINSEL0 &= ~(0x0F << 20); // Clear PINSEL BITS 20-23 (TX2 and RX2)
288  LPC_PINCON->PINSEL0 |= ((UART2_PINFUNCTION & 0x03) << 20); // Set TXD2 pin function
289  LPC_PINCON->PINSEL0 |= ((UART2_PINFUNCTION & 0x03) << 22); // Set RXD2 pin function
290 
291  LPC_PINCON->PINMODE0 &= ~(0x0F << 20); // Pin mode 00 (pull-up resistor) (for TX2 and RX2)
292 
293  LPC_PINCON->PINMODE_OD0 &= ~((1 << 10) | (1 << 11)); // Select normal mode (not open drain) for P0[10] and P0[11] (TX2 and RX2)
294 
295  LPC_SC->PCONP |= UART2_PCONP_ENABLE; // Enable UART2
296  LPC_SC->PCLKSEL1 &= ~(0x03 << 16); // Clean PCLK of UART2
297  LPC_SC->PCLKSEL1 |= (UART_PCLK_VAL << 16); // Set PCLK of UART2 to CCLK/2
298 
299  // FIFOs are empty
300  UARTx->FCR = (UART_FCR_FIFO_EN | UART_FCR_RX_RS | UART_FCR_TX_RS);
301  //UARTx->FCR = 0; // Disable FIFO
302 
303  while (UARTx->LSR & UART_LSR_RDR) { // Dummy reading - empty rx!
304  tmp = UARTx->RBR;
305  }
306  UARTx->TER = UART_TER_TXEN;
307  while (!(UARTx->LSR & UART_LSR_THRE)); // Wait for Tx complete
308  UARTx->TER = 0; // Disable Tx
309  UARTx->IER = 0; // Disable interrupt
310  UARTx->LCR = 0; // Set LCR to default state
311  UARTx->ACR = 0; // Set ACR to default state
312 
313  tmp = UARTx->LSR; // Clean status
314 
315  if (UART_SetDivisors(baud) < 0) return false;
316 
317  tmp = (UARTx->LCR & (UART_LCR_DLAB_EN | UART_LCR_BREAK_EN)) & UART_LCR_BITMASK;
318  tmp |= 0x03;
319 
320  UARTx->LCR = (uint8_t) (tmp & UART_LCR_BITMASK);
321  UARTx->TER |= UART_TER_TXEN;
322 
323  UARTx->FCR = UART_FCR_FIFO_EN | UART_FCR_TRG_LEV0;
324  UARTx->IER = UART_IER_RBRINT_EN | UART_IER_RLSINT_EN;
325  intrTxStatus = false;
326  RBUF_RESET(rbuffer.rxWrite);
327  RBUF_RESET(rbuffer.rxRead);
328  RBUF_RESET(rbuffer.txWrite);
329  RBUF_RESET(rbuffer.txRead);
330  while (1) {
331  uint32_t iir;
332  iir = (UARTx->IIR & 0x03CF) & UART_IIR_INTID_MASK;
333  if ((iir == UART_IIR_INTID_RDA) || (iir == UART_IIR_INTID_CTI)) {
334  tmp = UARTx->RBR;
335  }
336  else break;
337  }
338  if (!UART_IsEnabled(UART2_IRQn)) {
339  NVIC_EnableIRQ(UART2_IRQn);
340  }
341 
342  return true;
343 }
