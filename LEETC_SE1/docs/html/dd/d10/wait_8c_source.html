149 static int32_t WAIT_TIM0_Init(void) // Initialise Timer0
150 {
151  LPC_SC->PCONP |= PCONP_ENABLE_TIM0;
152  LPC_TIM0->CTCR = 0; // Set to timer mode
153 
154  LPC_SC->PCLKSEL0 &= ~(BITS_PCLK_TIM0); // Set PCLK to 1/4, "00" (bits[3:2] correspond to timer0)
155  LPC_TIM0->PR = TIMER_PCLK*TIMER_RES - 1; // Prescale for time resolution of 1 microsecond and pclk=cclk/4
156 
157  LPC_TIM0->MCR = 0x0; // Disable all match registers
158  LPC_TIM0->MCR |= 0x3; // Set MR0 to reset and interrupt
159 
160  LPC_TIM0->TCR = DISABLE_RESET;
161 
162  LPC_TIM0->IR = TIMER_IR_MR0; // Reset interrupt bit
163 
164  return 0;
165 }
