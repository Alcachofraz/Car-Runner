138 void NETWORK_ScorePublisherTask(void *pvParameters) {
139  unsigned char buffer[128];
140  MQTTTransport transporter;
141  MQTTString topicString;
142  int result;
143  int length;
144 
145  // ESP8266 Transport Layer
146  static transport_iofunctions_t iof = { NETWORK_Send, NETWORK_Recv };
147  int transport_socket = transport_open(&iof);
148 
149  int state = NETWORK_STATE_IDLE;
150  for (;;) {
151  switch (state) {
152  case NETWORK_STATE_IDLE:
153  xQueueReceive(queuePUBLISH_SCORE, &score, portMAX_DELAY);
154  ESP_Take();
155  state++;
156  case NETWORK_STATE_INIT:
157  if (NETWORK_Connect(MQTT_ADDRESS, MQTT_PORT, CONNECTION_KEEPALIVE_S)) {
158  state++;
159  }
160  break;
161  case NETWORK_STATE_CONNECT:
162  transporter.sck = &transport_socket;
163  transporter.getfn = transport_getdatanb;
164  transporter.state = 0;
165 
166  MQTTPacket_connectData connectData = MQTTPacket_connectData_initializer;
167  connectData.MQTTVersion = 3;
168  connectData.username.cstring = MQTT_DEVICE_TOKEN;
169  connectData.clientID.cstring = MQTT_DEVICE_TOKEN;
170  connectData.keepAliveInterval = CONNECTION_KEEPALIVE_S * 2;
171  length = MQTTSerialize_connect(buffer, sizeof(buffer), &connectData);
172  // Send CONNECT to the mqtt broker.
173  if ((result = transport_sendPacketBuffer(transport_socket, buffer, length)) == length) {
174  state++;
175  } else {
176  state = NETWORK_STATE_INIT;
177  }
178  break;
179  case NETWORK_STATE_WAIT_CONNECT:
180  // Wait for CONNACK response from the MQTT broker.
181  while (true) {
182  if ((result = MQTTPacket_readnb(buffer, sizeof(buffer), &transporter)) == CONNACK) {
183  // Check if the connection was accepted.
184  unsigned char sessionPresent, connack_rc;
185  if ((MQTTDeserialize_connack(&sessionPresent, &connack_rc, buffer,
186  sizeof(buffer)) != 1) || (connack_rc != 0)) {
187  state = NETWORK_STATE_INIT;
188  break;
189  } else {
190  state++;
191  break;
192  }
193  } else if (result == -1) {
194  state = NETWORK_STATE_INIT;
195  break;
196  }
197  }
198  break;
199  case NETWORK_STATE_PUBLISH:
200  MQTT_GetInitialiser(&topicString);
201  topicString.cstring = "v1/devices/me/telemetry";
202  unsigned char payload[64];
203  length = sprintf((char *) payload, "{\"group\":%d, \"score\":%d}", GROUP_ID, score);
204  length = MQTTSerialize_publish(buffer, sizeof(buffer), 0, 0, 0, 0, topicString, payload, length);
205  // Send PUBLISH to the MQTT broker.
206  if ((result = transport_sendPacketBuffer(transport_socket, buffer, length)) == length) {
207  state++;
208  } else {
209  state = NETWORK_STATE_INIT;
210  }
211  break;
212  case NETWORK_STATE_DISCONNECT:
213  if (ESP_Close() == 0) state = NETWORK_STATE_IDLE;
214  ESP_Give();
215  default:
216  state = NETWORK_STATE_IDLE;
217  }
218  }
219 
220  vTaskDelete(NULL);
221 }
