58  SPI_Init();

64  LPC_PINCON->PINSEL1 &= ~(0x03 << 0); // CS
65  LPC_GPIO0->FIODIR |= (1 << CS); // Output
66 
67  // Set !CS to high (i.e. disable CS):

69

110  txBuffer[0] = (ADXL_READ_BIT | ADXL_MB_BIT | DATAX0);
