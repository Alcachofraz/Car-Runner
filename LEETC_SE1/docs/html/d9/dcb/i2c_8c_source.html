38  switch ((LPC_SC->PCLKSEL0 >> 6) & 0x03) {
39  case 0x00:
40  return SystemCoreClock/4;
41  case 0x01:
42  return SystemCoreClock;
43  case 0x02:
44  return SystemCoreClock/2;
45  case 0x03:
46  return SystemCoreClock/8;
47  default:

57 void I2C1_IRQHandler(void) { // I2C1 interrupt handler
58 
59  switch (LPC_I2C1->I2STAT) {
60  case I2C_START: // Start
61  I2C1WriteIndex = 0;
62  LPC_I2C1->I2DAT = I2C1WriteBuffer[I2C1WriteIndex++];
63  LPC_I2C1->I2CONCLR = STA;
64  I2C1State = I2C_BUSY;
65  break;
66 
67  case I2C_REPEATED_START: // Repeated Start
68  I2C1ReadIndex = 0;
69  LPC_I2C1->I2DAT = I2C1WriteBuffer[I2C1WriteIndex++];
70  LPC_I2C1->I2CONCLR = STA;
71  I2C1State = I2C_BUSY;
72  break;
73 
74  case I2C_SLAW_ACK: // [SLA + W] transmitted and ACK received
75  if (I2C1WriteLength == 1) { // There's no data to send
76  LPC_I2C1->I2CONSET = STO; // Set STOP flag
77  I2C1State = I2C_IDLE;
78  }
79  else {
80  LPC_I2C1->I2DAT = I2C1WriteBuffer[I2C1WriteIndex++];
81  }
82  break;
83 
84  case I2C_SLAW_NACK: // [SLA + W] transmitted and NACK received
85  LPC_I2C1->I2CONSET = STO;
86  I2C1State = I2C_IDLE;
87  break;
88 
89  case I2C_DATAW_ACK: // Data byte transmitted, ACK received
90  if (I2C1WriteIndex < I2C1WriteLength) {
91  LPC_I2C1->I2DAT = I2C1WriteBuffer[I2C1WriteIndex++]; // Keep transmitting
92  }
93  else { // There's nothing more to transmit
94  if (I2C1ReadLength != 0) { // There's something to receive
95  LPC_I2C1->I2CONSET = STA; // Repeated start for read operation
96  }
97  else {
98  LPC_I2C1->I2CONSET = STO;
99  I2C1State = I2C_IDLE;
100  }
101  }
102  break;
103 
104  case I2C_DATAW_NACK: // Data byte transmitted, NACK received
105  LPC_I2C1->I2CONSET = STO;
106  I2C1State = I2C_IDLE;
107  break;
108 
109  case I2C_ARBITRATION_LOST: // Arbitration lost. This API doesn't handle with multiple Master operations.
110  LPC_I2C1->I2CONSET = STO;
111  I2C1State = I2C_IDLE;
112  break;
113 
114  case I2C_SLAR_ACK: // [SLA + R] transmitted and ACK received
115  if (I2C1ReadLength > 0) {
116  // Will go to I2C_DATAR_ACK
117  LPC_I2C1->I2CONSET = AA; // Assert ACK
118  }
119  else {
120  // Will go to I2C_DATAR_NACK
121  LPC_I2C1->I2CONCLR = AA; // Assert NACK
122  }
123  break;
124 
125  case I2C_SLAR_NACK: // [SLA + R] transmitted and NACK received.
126  LPC_I2C1->I2CONSET = STO;
127  I2C1State = I2C_IDLE;
128  break;
129 
130  case I2C_DATAR_ACK: // Data byte received, ACK returned
131  I2C1ReadBuffer[I2C1ReadIndex++] = LPC_I2C1->I2DAT;
132  if (I2C1ReadIndex < I2C1ReadLength) {
133  LPC_I2C1->I2CONSET = AA; // Assert ACK
134  }
135  else {
136  LPC_I2C1->I2CONCLR = AA; // Assert NACK
137  }
138  break;
139 
140  case I2C_DATAR_NACK: // Data byte received, NACK returned
141  I2C1ReadBuffer[I2C1ReadIndex++] = LPC_I2C1->I2DAT;
142  LPC_I2C1->I2CONSET = STO;
143  I2C1State = I2C_IDLE;
144  break;
145 
146  default:
147  LPC_I2C1->I2CONSET = STO;
148  I2C1State = I2C_IDLE;
149  break;
150  }
151  LPC_I2C1->I2CONCLR = SI;
152 }

243 void I2C1_Init() {
244  WAIT_Init(SYS);
245 
246  LPC_SC->PCONP |= I2C1_PCONP_ENABLE;
247 
248  LPC_SC->PCLKSEL1 &= ~(0x03 << 6); // Set PCLK to CCLK/4
249 
250  LPC_PINCON->PINSEL1 &= ~(0x0F << 6); // Clear SDA1 and SCL1 PINSEL BITS
251  LPC_PINCON->PINSEL1 |= ((I2C1_PINSEL & 0x03) << 6); // SDA1
252  LPC_PINCON->PINSEL1 |= ((I2C1_PINSEL & 0x03) << 8); // SCL1
253 
254  LPC_PINCON->PINMODE1 &= ~((0x3<<6)|(0x3<<8));
255  LPC_PINCON->PINMODE1 |= ((0x2<<6)|(0x2<<8));
256 
257  LPC_PINCON->PINMODE_OD0 |= ((0x1<<19)|(0x1<<20));
258 
259  if (!I2C_IsEnabled(I2C1_IRQn)) { // If it's not already enabled:
260  NVIC_EnableIRQ(I2C1_IRQn); // Enable interrupts for I2C1 interface
261  }
262 
263  LPC_I2C1->I2CONCLR = AA | SI | STA | I2EN; // Clear control flags (STOP flag never needs clearing)
264 
265  LPC_I2C1->I2CONSET = I2EN; // Enable interrupts
266 }
