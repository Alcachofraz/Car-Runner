171 uint32_t ESP_Ip(unsigned char* ip) {
172  ESP_WriteString("AT+CIFSR");
173  ESP_WriteString(AT_CMD_SUFFIX);
174  char received;
175  do {
176  if (!ESP_ReadChar(&received, ESP_TIMEOUT_MS)) return -1;
177  } while (received < '0' || received > '9');
178  for (unsigned char i = 0; i < 4; i++) {
179  ip[i] = 0;
180  do {
181  ip[i] = 10 * ip[i] + received - '0';
182  if (!ESP_ReadChar(&received, ESP_TIMEOUT_MS)) return -1;
183  } while (received >= '0' && received <= '9');
184  if (!ESP_ReadChar(&received, ESP_TIMEOUT_MS)) return -1;
185  }
186  return ESP_WaitForString(ESP_TIMEOUT_MS, 1, "OK");
187 }

252 uint32_t ESP_ReceiveData(char * data, uint32_t max_len) {
253  char received;
254  int bytes = 0; // Current 'data' receiving index
255  char len_str[5]; // Max of 4 characters (max length is 2048 for an IPD packet)
256  do {
257  if (len == 0) {
258  // Wait for +IPD
259  ESP_WaitFor("+IPD,", ESP_LONG_TIMEOUT_MS);
260 
261  // Read length:
262  int i;
263  for (i = 0;; i++) {
264  if (!ESP_ReadChar(&received, ESP_TIMEOUT_MS)) return -1;
265  if (received == ':') break;
266  len_str[i] = received;
267  }
268  len_str[i] = '\0';
269  char * dummy;
270  len = strtol(len_str, &dummy, 10);
271  }
272 
273  // Read to data 'len' bytes, or until 'max_len' is met:
274  while (len > 0) {
275  if (!ESP_ReadChar(&data[bytes], ESP_TIMEOUT_MS)) return -1;
276  len--;
277  if (bytes++ >= max_len - 1) break;
278  }
279  } while (max_len > bytes); // While max_len hasn't been met
280 
281  return bytes;
282 }

284 uint32_t ESP_WaitForString(uint32_t timeout, int resNum, ...) {
285  va_list valist;
286  va_start(valist, resNum);
287 
288  char * res[resNum];
289 
290  for (int i = 0; i < resNum; i++) {
291  res[i] = va_arg(valist, char*);
292  }
293 
294  while (ESP_ReadString(AT_RECEIVING_BUFFER, timeout) > 0) {
295  for (int i = 0; i < resNum; i++) {
296  if (strcmp((char *) AT_RECEIVING_BUFFER, res[i]) == 0) {
297  va_end(valist);
298  return i;
299  }
300  }
301  }
302  va_end(valist);
303  return -1;

306 uint32_t ESP_WaitFor(char * response, uint32_t timeout) {
307  uint32_t bytes = 0;
308  int i = 0;
309  char ch;
310  do {
311  if (!ESP_ReadChar(&ch, timeout)) break;
312  AT_RECEIVING_BUFFER[bytes] = ch;
313  bytes++;
314  if (ch == response[i]) {
315  i++;
316  }
317  else i = 0;
318  } while (response[i] != 0);
