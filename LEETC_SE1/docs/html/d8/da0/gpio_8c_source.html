38  void GPIO_PinFunction(uint8_t v_pinNumber_u8, uint8_t v_pinFunction_u8)
39  ****************************************************************************************************
40  * I/P Arguments:
41  uint8_t: pin number for which the pin function needs to be selected.
42  .
43  v_pinFunction_u8: Any of the below four pins functions.
44  PINSEL_FUNC_0
45  PINSEL_FUNC_1
46  PINSEL_FUNC_2
47  PINSEL_FUNC_3
48  Refer lpc1768 data sheet for more info.
49  * Return value : none

51  Refer the lpc1768 data sheet for complete pin functions.
52  ****************************************************************************************************/
53 void GPIO_PinFunction(uint8_t v_pinNumber_u8, uint8_t v_pinFunction_u8)
54 {
55  uint8_t v_portNumber_u8;
56  uint32_t *ptr_PINCON;
57 
58  if((v_pinNumber_u8 < P_MAX) && (v_pinFunction_u8 <= PINSEL_FUNC_3)) //Handle the request only if it is in range
59  {
60  v_portNumber_u8 = (v_pinNumber_u8>>5); //Divide the pin number by 32 go get the PORT number
61  v_pinNumber_u8 = v_pinNumber_u8 & 0x1f; //lower 5-bits contains the bit number of a 32bit port
62 
63 
64  v_portNumber_u8 = v_portNumber_u8*2;
65  if(v_pinNumber_u8>=16)
66  {
67  v_portNumber_u8++;
68  v_pinNumber_u8 = v_pinNumber_u8-16;
69  }
70 
71  v_pinNumber_u8 = v_pinNumber_u8 * 2;
72 
73  ptr_PINCON = ((uint32_t *)&LPC_PINCON->PINSEL0 + v_portNumber_u8);
74 
75  *(uint32_t *)(ptr_PINCON) &= ~(0x03UL << v_pinNumber_u8);
76  *(uint32_t *)(ptr_PINCON) |= (((uint32_t)v_pinFunction_u8) << v_pinNumber_u8);
77  }
78 }
