19 static uint32_t SPI_GetPCLK(void) {
20  switch ((LPC_SC->PCLKSEL0 >> 16) & 0x03) {
21  case 0x00:
22  return SystemCoreClock/4;
23  case 0x01:
24  return SystemCoreClock;
25  case 0x02:
26  return SystemCoreClock/2;
27  case 0x03:
28  return SystemCoreClock/8;
29  default:
30  return -1;
31  }
32 }

46 int32_t SPI_ConfigTransfer(int frequency, int bitData, int mode) {
47  // Validate bitData:
48  if ((bitData < 8) || (bitData > 16)) return -1;
49 
50  // PCLK cycles that make up an SPI clock:
51  LPC_SPI->SPCCR = ((SPI_GetPCLK()/frequency) & SPI_CCR_MASK);
52 
53  // SPI Control register:
54  LPC_SPI->SPCR = 0; // Clear everything
55  LPC_SPI->SPCR |= (((bitData << 8) & SPI_CR_BITS_MASK) | MSTR | ((mode & 0x03) << 3) | BIT_ENABLE);
56 
57  return 0;
58 }

60 int32_t SPI_Transfer(unsigned short *txBuffer, unsigned short *rxBuffer, int length) {
61  for (int i=0; i<length; i++) {
62  // Write:
63  LPC_SPI->SPDR = txBuffer[i];
64  while ((LPC_SPI->SPSR & SPIF_MASK) == 0) { // Wait for transfer completed flag
65  if ((LPC_SPI->SPSR & SPI_STATUS_ERROR_MASK) != 0) return -1;
66  }
67  // Read:
68  rxBuffer[i] = LPC_SPI->SPDR;
69  }
70  return 0;
71 }
