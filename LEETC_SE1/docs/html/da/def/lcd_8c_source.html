104  column = (column > (LCD_DDRAM_LENGTH * LCD_DISPLAY_ROWS) || column < 1)

157  WAIT_TIM3_Us(200);

160  LCD_WriteCommand(OFF); // Turn off display and cursor
161  LCD_Clear();

163  LCD_WriteCommand(ON); // Turn on display, cursor and blinking of cursor
164  LCD_Locate(1, 1);

174  return -1;
175  }

178  return 0;
