148 static int getDaysInMonth() {
149  int february_days = 28;
150  int year = LPC_RTC->YEAR;
151  int month = LPC_RTC->MONTH + 1;
152 
153  if ((year%4 == 0) && (year%100 != 0)) {
154  february_days = 29;
155  }
156 
157  if (month == 2) return february_days;
158  else if (month == 1 || month == 3 || month == 5 || month == 8 || month == 10 || month == 12) return 31;
159  else return 30;
