698 void idle(void) {
699 
700  int volatile side = 0; // 0 to show time, 1 to show best score
701  int volatile push_events = 0; // Buttons events
702 
703  BUTTON_WaitRelease(idleWait);
704 
705  while (1) {
706  Score score;
707 
708  // State Machine:
709  switch (side) {
710  case 0: // Showing time
711  printDateTime(NONE);
712  break;
713  case 1: // Showing best score
714  SCORE_Get(&score, scoreCount);
715  printScore(&score, scoreCount);
716  break;
717  }
718 
719  // Input Handler:
720  if ((push_events = BUTTON_GetButtonsPushEvents(&bitmap)) != 0) {
721  if (push_events & B1) { // Show time
722  if (side != 0) {
723  side = 0;
724  LCDText_Clear();
725  }
726  }
727  else if (push_events & B2) { // Show best score
728  if (side != 1) {
729  side = 1;
730  LCDText_Clear();
731  }
732  }
733  else if (push_events & B3) { // Start a game
734  LCDText_Clear();
735  state = STATE_PREGAME;
736  return;
737  }
738  }
739  if ((bitmap & B1) && (bitmap & B2)) { // Check for buttons hold, and enter Configuration Mode
740  LCDText_Clear();
741  LCDText_Locate(1, 1);
742  LCDText_Printf("Entering config");
743  LCDText_Locate(2, 1);
744  LCDText_Printf("mode...");
745 
746  uint32_t start = WAIT_SYS_GetElapsedMs(0);
747  bool exit = false;
748  while (WAIT_SYS_GetElapsedMs(start) < BUTTONS_HOLD_TIME) {
749  if (!(BUTTON_Hit() & B1) || !(BUTTON_Hit() & B2)) { // If any of the buttons were released
750  exit = true;
751  break;
752  }
753  }
754 
755  if (exit) continue;
756 
757  state = STATE_CONFIG; // If user held the buttons until this point, enter Configuration Mode
758  return;
759  }
760  }
761 }

1217 void vConfigureTimerForRunTimeStats(void) {

1219 
1220  /* This function configures a timer that is used as the time base when
1221  collecting run time statistical information - basically the percentage
1222  of CPU time that each task is utilising. It is called automatically when
1223  the scheduler is started (assuming configGENERATE_RUN_TIME_STATS is set
1224  to 1). */
1225

1232 
1233  /* Just count up. */
1234  LPC_TIM1->CTCR = CTCR_CTM_TIMER;

1237  but not too fast so as to overflow all the time. */
