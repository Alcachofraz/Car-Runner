43 
44  bool done = false;
45  int lowestScore = 0;
46 
47  for (int i = 0; i < SCORE_NUM; i++) {
48  newScores[i] = oldScores[i];
49  if ((strncmp(newScores[i].name, username, NAME_LENGTH+1) == 0) && !done) { // If user name coincides...
50  if (newScores[i].score < score) { // If the new score is better...
51  newScores[i].score = score;
52  done = true;
53  }
54  else return; // If not, return (no need to update flash)
55  }
56  else if ((newScores[i].score == 0) && !done) { // If there is space available...
57  strncpy(newScores[i].name, username, NAME_LENGTH+1);
58  newScores[i].score = score;
59  done = true;
60  }
61  if (newScores[i].score < newScores[lowestScore].score) {
62  lowestScore = i;
63  }
64  }
65 
66  if (!done) { // If user name was not found, and there is no space available...
67  if (score > newScores[lowestScore].score) { // If score is higher than the lowest registered...
68  strncpy(newScores[lowestScore].name, username, NAME_LENGTH+1);
69  newScores[lowestScore].score = score;
70  }
71  }

73  sortScores(newScores, SCORE_NUM);
