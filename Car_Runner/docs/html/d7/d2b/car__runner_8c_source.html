106 void getUsername(void) {
107 
108  LCDText_Locate(1, 1);
109  LCDText_Printf("User name:");
110 
111  off = false;
112  int volatile events = 0; // Buttons events
113  int ch = 0; // [0, 10] -> Character selected, 11 is the special character '\0'
114  char aux[NAME_LENGTH+1]; // Auxiliary variable for blinking
115 
116  while (1) {
117  if (!WAIT_IsBusy(TIM0)) WAIT_IRQ0_Us(HIGHLIGHT_BLINK_TIME, blinkHandler); // If it's not busy, start counting
118 
119  LCDText_Locate(2, 1);
120  if (off) {
121  strcpy(aux, username);
122  aux[ch] = ' ';
123  LCDText_Printf(aux);
124  }
125  else LCDText_Printf(username);
126 
127  if ((events = BUTTON_GetButtonsEvents()) != 0) {
128  if ((events & B1) && !(BUTTON_Hit() & B1)) { // Increment
129  switch (username[ch]) {
130  case 32:
131  username[ch] = 97;
132  break;
133  case 122:
134  username[ch] = 32;
135  break;
136  default:
137  username[ch]++;
138  break;
139  }
140  }
141  else if ((events & B2) && !(BUTTON_Hit() & B2)) { // Decrement
142  switch (username[ch]) {
143  case 32:
144  username[ch] = 122;
145  break;
146  case 97:
147  username[ch] = 32;
148  break;
149  default:
150  username[ch]--;
151  break;
152  }
153  }
154  else if ((events & B3) && !(BUTTON_Hit() & B3)) { // Confirm value
155  if (ch == 0) username[0] -= 32; // First character must be upper case
156  if (++ch > (NAME_LENGTH-1)) { // If it's the last character
157  break; // Get out
158  }
159  }
160  }
161  }
162  while (WAIT_IsBusy(TIM0)); // Wait for timer0
163 }

302 void idle(void) {
303  currentScore = 0;
304 
305  int volatile side = 0; // 0 to show time, 1 to show bets score
306  int volatile events = 0; // Buttons events
307 
308  BUTTON_WaitRelease(idleWait);
309 
310  while (1) {
311  Score bestScore;
312 
313  // State Machine:
314  switch (side) {
315  case 0: // Showing time
316  updateAndPrintTime(NONE);
317  break;
318  case 1: // Showing best score
319  if (!WAIT_IsBusy(TIM1)) WAIT_IRQ1_Us(SCORE_SHOW_TIME, currentScoreHandler); // Call 'currentScoreHandler()' in 'SCORE_SHOW_TIME' microseconds
320  getScore(&bestScore, currentScore);
321  printScore(&bestScore, currentScore);
322  break;
323  }
324 
325  // Input Handler:
326  if ((events = BUTTON_GetButtonsEvents()) != 0) {
327  if ((events & B1) && !(BUTTON_Hit() & B1)) { // Show time
328  if (side != 0) {
329  side = 0;
330  LCDText_Clear();
331  }
332  }
333  else if ((events & B2) && !(BUTTON_Hit() & B2)) { // Show best score
334  if (side != 1) {
335  side = 1;
336  LCDText_Clear();
337  }
338  }
339  else if ((events & B3) && !(BUTTON_Hit() & B3)) { // Start a game
340  LCDText_Clear();
341  while (WAIT_IsBusy(TIM0) || WAIT_IsBusy(TIM1)) { // Wait for Timer0 and Timer1 before starting game
342  LCDText_Locate(1, 1);
343  LCDText_Printf("Starting game...");
344  }
345  state = PREGAME;
346  return;
347  }
348  else if ((BUTTON_Hit() & B1) && (BUTTON_Hit() & B2)) { // Check for buttons hold, and enter Configuration Mode
349  if (WAIT_IsBusy(TIM0)) continue; // If previous Timer0 interrupt hasn't arrived, get out
350 
351  LCDText_Clear();
352  LCDText_Locate(1, 1);
353  LCDText_Printf("Entering config");
354  LCDText_Locate(2, 1);
355  LCDText_Printf("mode...");
356 
357  WAIT_IRQ0_Us(BUTTONS_HOLD_TIME, dummy); // Interrupt in BUTTONS_HOLD_TIME microseconds
358  while (WAIT_IsBusy(TIM0)) {
359  if (!(BUTTON_Hit() & B1) || !(BUTTON_Hit() & B2)) { // If any of the buttons were released
360  break;
361  }
362  }
363  if (!WAIT_IsBusy(TIM0)) {
364  state = CONFIG; // If user held the buttons until this point, enter Configuration Mode
365  return;
366  }
367  LCDText_Clear();
368  }
369  }
370  }
371 }

486 void randomiseFuel(int from, int to, int map[LCD_DISPLAY_ROWS][LCD_DDRAM_LENGTH]) {
487  int gaps[LCD_DDRAM_LENGTH] = {0};
488  int ct = 0;
489  for (int i = from; i <= to; i++) {
490  if ((map[0][i-1] == 0) && (map[1][i-1] == 0)) {
491  gaps[ct++] = i-1;
492  }
493  }
494  srand(RTC_GetSeconds()); // Set randomiser seed to RTC time_t value
495 
496  int col = gaps[rand()%ct];
497  int row = rand()%2;
498 
499  // Write obstacle in LCD:
500  LCDText_Locate(row+1, col+1);
501  LCDText_WriteChar(FUEL_CHAR);
502 
503  map[row][col] = 2;
504 }

506 void randomiseObstacles(int from, int to, int map[LCD_DISPLAY_ROWS][LCD_DDRAM_LENGTH]) {
507  for (int i = from; i <= to; i++) {
508  // Erase obstacles from array of obstacles:
509  map[0][i-1] = 0;
510  map[1][i-1] = 0;
511 
512  // Erase obstacles from LCD:
513  LCDText_Locate(1, i);
514  LCDText_WriteChar(' ');
515  LCDText_Locate(2, i);
516  LCDText_WriteChar(' ');
517  }
518  srand(RTC_GetSeconds()); // Set randomiser seed to RTC time_t value
519 
520  // Randomise first column:
521  int col = from;
522  col += rand()%3+3; // Always leave a space of 2-4 columns between each obstacle
523 
524  int row;
525 
526  while(col <= to) {
527  // Randomise row:
528  row = rand()%2+1;
529 
530  // Write obstacle in LCD:
531  LCDText_Locate(row, col);
532  LCDText_WriteChar(BARRIER_CHAR);
533  // Write obstacle in array of obstacles:
534  map[row-1][col-1] = 1;
535 
536  // Randomise next column:
537  col += rand()%3+3; // Always leave a space of 2-4 columns between each obstacle
538  }
539 }

567 void game(void) {
568 
569  // Car:
570  CAR car = {.row = 1, .last_row = 1, .front_column = CAR_POSITION + 1, .back_column = CAR_POSITION};
571 
572  // Array that represents all current obstacles in LCD DDRAM
573  int map[LCD_DISPLAY_ROWS][LCD_DDRAM_LENGTH] = {0};
574 
575  int currentFuelCol = 1;
576 
577  points = 1;
578  fuel = MAX_FUEL;
579 
580  randomiseObstacles(INITIAL_OBSTACLE_GAP, LCD_DDRAM_LENGTH, map); // Randomise obstacles
581  randomiseFuel(INITIAL_OBSTACLE_GAP, LCD_DDRAM_LENGTH, map); // Randomise fuel galleons
582 
583  while(1) {
584  changeRow(&car, getInclination());
585 
586  if (!WAIT_IsBusy(TIM0)) {
587  // Refresh obstacles and fuel galleons:
588  if (car.back_column == CAR_POSITION + (LCD_DDRAM_LENGTH/2)) { // If display is showing [21, 36]
589  // Refresh first 20:
590  randomiseObstacles(1, LCD_DDRAM_LENGTH/2, map);
591  randomiseFuel(1, LCD_DDRAM_LENGTH/2, map);
592  }
593  if (car.back_column == CAR_POSITION) { // If display is showing [1, 16]
594  // Refresh last 20:
595  randomiseObstacles((LCD_DDRAM_LENGTH/2) + 1, LCD_DDRAM_LENGTH, map);
596  randomiseFuel((LCD_DDRAM_LENGTH/2) + 1, LCD_DDRAM_LENGTH, map);
597  }
598 
599  // Shift car:
600  shiftCar(&car);
601 
602  // Shift fuel indicator:
603  shiftFuelIndicator(&currentFuelCol);
604 
605  // Shift display:
606  LCDText_ShiftDisplay(LEFT);
607 
608  // Verify if car hit an obstacle:
609  if (checkForLoss(&car, map)) break; // Lose
610 
611  checkForFuelGrab(&car, map);
612 
613  // Game delay:
614  WAIT_IRQ0_Us(GAME_RATE, dummy);
615  }
616  if (!WAIT_IsBusy(TIM1)) WAIT_IRQ1_Us(FUEL_RATE, busyPointsHandler); // Points delay
617  }
618 }

650 void config(void) {

652  LCDText_Printf("> Time Config");
653  LCDText_Locate(2, 1);
654  LCDText_Printf(" Score Config");
655 
656  BUTTON_WaitRelease(0);
657 
658  int volatile events = 0; // Buttons events
659  int volatile last_option = 0; // Last selected option
660  int volatile option = 0; // Current selected option
661 
662  while (1) {
663  // State Machine:
664  switch (option) {
665  case 0: // Time Configuration option highlighted
666  LCDText_Locate(1, 1);
667  LCDText_Printf("> Time Config ");
668  LCDText_Locate(2, 1);
669  LCDText_Printf(" Score Config");
670  break;
671  case 1: // Best Score Configuration option highlighted
672  switch(last_option) {
673  case 0:
674  LCDText_Locate(1, 1);
675  LCDText_Printf(" Time Config ");
676  LCDText_Locate(2, 1);
677  LCDText_Printf("> Score Config");
678  break;
679  case 2:
680  LCDText_Locate(1, 1);
681  LCDText_Printf("> Score Config");
682  LCDText_Locate(2, 1);
683  LCDText_Printf(" Name Config ");
684  break;
685  }
686  break;
687  case 2:
688  LCDText_Locate(1, 1);
689  LCDText_Printf(" Score Config");
690  LCDText_Locate(2, 1);
691  LCDText_Printf("> Name Config ");
692  }
693 
694  // Input Handler:
695  if ((events = BUTTON_GetButtonsEvents()) != 0) {
696  if ((events & B1) && !(BUTTON_Hit() & B1)) { // Go to upper option
697  last_option = option;
698  if (++option > (CONFIG_OPTIONS-1)) option = CONFIG_OPTIONS-1;
699  }
700  else if ((events & B2) && !(BUTTON_Hit() & B2)) { // Go to lower option
701  last_option = option;
702  if (--option < 0) option = 0;
703  }
704  else if ((events & B3) && !(BUTTON_Hit() & B3)) { // Select option
705  switch (option) {
706  case 0: // Time Configuration:
707  while(WAIT_IsBusy(TIM0)); // Wait for Timer0
708  state = CONFIG_DATE;
709  return;
710  case 1: // Score Configuration:
711  while(WAIT_IsBusy(TIM0)); // Wait for Timer0
712  state = CONFIG_ERASE_SCORES;
713  return;
714  case 2: // Name Configuration:
715  while(WAIT_IsBusy(TIM0)); // Wait for Timer0
716  username[0] += 32; // Lower case
717  state = CONFIG_NAME;
718  return;
719  }
720  }
721  else if ((BUTTON_Hit() & B1) && (BUTTON_Hit() & B2)) { // Check for buttons hold, and go back to menu
722  if (WAIT_IsBusy(TIM0)) continue; // If previous interrupt hasn't arrived, get out
723

725  LCDText_Locate(1, 1);
726  LCDText_Printf("Returning to");
727  LCDText_Locate(2, 1);
728  LCDText_Printf("menu...");
729 
730  WAIT_IRQ0_Us(BUTTONS_HOLD_TIME, dummy); // Interrupt in 2 seconds
731  while (WAIT_IsBusy(TIM0)) {
732  if (!(BUTTON_Hit() & B1) || !(BUTTON_Hit() & B2)) { // If any of the buttons were released
733  break;
734  }
735  }
736  if (!WAIT_IsBusy(TIM0)) { // If user held the buttons until this point, return to Idle Mode
737  state = IDLE;
738  return;
739  }
740  LCDText_Clear();
741  }
742  }
743  }
744 }

757 void timeConfig(void) {
758 
759  RTC_Disable(); // Disable RTC counters, to stop the clock while adjusting it
760 
761  int volatile events = 0; // Buttons events
762 
763  int field = 0; // [0, 5] -> Every time field, excluding day of year and seconds
764  RTC_TIME_FIELD timeFields[TIME_CONFIG_FIELDS] = {YEAR, MONTH, DOM, DOW, HOUR, MIN};
